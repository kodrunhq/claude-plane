3. Create `web/src/api/client.ts`:
   - `export class ApiError extends Error { constructor(public status: number, message: string) }`
   - `export async function request<T>(path: string, options?: RequestInit): Promise<T>`:
     - Base URL: `/api/v1`
     - Reads token from `localStorage.getItem('token')`
     - Sets `Content-Type: application/json` and `Authorization: Bearer {token}` headers
     - On non-ok response: parse JSON for error message, throw `ApiError`
     - On ok: return `res.json()`

4. Refactor `web/src/api/sessions.ts` to use the new `request()` helper from `client.ts` instead of inline fetch. Preserve the same exports (createSession, listSessions, getSession, terminateSession). Wrap them in a `sessionsApi` object for consistency:
   ```typescript
   export const sessionsApi = {
     list: (filters?: Record<string, string>) => request<Session[]>(`/sessions${filters ? `?${new URLSearchParams(filters)}` : ''}`),
     get: (id: string) => request<Session>(`/sessions/${id}`),
     create: (params: CreateSessionRequest) => request<Session>('/sessions', { method: 'POST', body: JSON.stringify(params) }),
     kill: (id: string) => request<void>(`/sessions/${id}`, { method: 'DELETE' }),
   };
   // Keep named exports for backward compatibility with Phase 4 code
   export const { list: listSessions, get: getSession, create: createSession, kill: terminateSession } = sessionsApi;
   ```

5. Create `web/src/api/machines.ts`:
   ```typescript
   import { request } from './client';
   import type { Machine } from '../lib/types';

   export const machinesApi = {
     list: () => request<Machine[]>('/machines'),
     get: (id: string) => request<Machine>(`/machines/${id}`),
   };
   ```

   export const useUIStore = create<UIStore>((set) => ({
     sidebarCollapsed: false,
     toggleSidebar: () => set((s) => ({ sidebarCollapsed: !s.sidebarCollapsed })),
   }));
